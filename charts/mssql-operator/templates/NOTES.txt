Create your first Database CR:

  cat <<EOF | kubectl apply -f -
  apiVersion: mssql.popul.io/v1alpha1
  kind: Database
  metadata:
    name: my-database
  spec:
    server:
      host: mssql.example.com
      credentialsSecret:
        name: sa-credentials
    databaseName: MyDatabase
  EOF

List all MSSQL resources:

  kubectl get mssql
