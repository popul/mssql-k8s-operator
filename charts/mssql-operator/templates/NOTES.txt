Quick start -- deploy a managed SQL Server:

  # Create secrets
  kubectl create secret generic mssql-sa-password \
    --from-literal=MSSQL_SA_PASSWORD='YourStr0ngP@ssword!'
  kubectl create secret generic sa-credentials \
    --from-literal=username=sa --from-literal=password='YourStr0ngP@ssword!'

  # Deploy a SQL Server instance
  cat <<EOF | kubectl apply -f -
  apiVersion: mssql.popul.io/v1alpha1
  kind: SQLServer
  metadata:
    name: mssql
  spec:
    credentialsSecret:
      name: sa-credentials
    instance:
      acceptEULA: true
      saPasswordSecret:
        name: mssql-sa-password
      storageSize: 10Gi
  EOF

  # Wait for it
  kubectl get sqlsrv mssql -w

  # Create a database
  cat <<EOF | kubectl apply -f -
  apiVersion: mssql.popul.io/v1alpha1
  kind: Database
  metadata:
    name: myapp-db
  spec:
    server:
      sqlServerRef: mssql
    databaseName: myapp
  EOF
